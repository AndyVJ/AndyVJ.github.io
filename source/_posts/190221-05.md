```C
#include <reg52.h>

void delay10ms(unsigned int n);

//0~15 编码
unsigned char code num[]={0x3F, 0x06, 0x5B, 0x4F, 0x66, 0x6D, 0x7D, 0x07,

                          0x7F, 0x6F, 0x77, 0x7C, 0x39, 0x5E, 0x79, 0x71};

void main(void) 
{
    unsigned char i;

    while(1) {
        for (i = 0; i <= 15; i++) {
            P0 = ~num[i]; // 共阳极，取反
            delay10ms(50);
        }
    }

    return;
}

/*
 * 延时 10 ms 子函数
 */

void delay10ms(unsigned int n) 
{
    unsigned char a, b;

    for (; n > 0; n--) {
        for (a = 38; a > 0; a--) {
            for (b = 130; b > 0; b--);
        }
    }
}
```
